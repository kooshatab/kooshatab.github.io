#Load the data
data("maxtemp")

# Read about the dataset
?maxtemp

# Plot the full dataset
plot(maxtemp, ylab = "Maximum Annual Temperature (°C)", xlab = "Year", type = "o")


# Subset data from 1990 onward
maxtemp_1990_All = window(maxtemp, start = 1990)
maxtemp_1990 = window(maxtemp, start = 1990,end=2011)

# Fit a Simple Exponential Smoothing (SES) model to forecast 5 years ahead
fit_ses = ses(maxtemp_1990, h = 5)
# Fit a damped Holt’s linear trend model with optimal initialization
fit_holt = holt(maxtemp_1990,damped = TRUE, initial = "optimal",h = 5)
# the plot
plot(maxtemp_1990_All, ylab = "Maximum Annual Temperature (°C)", xlab = "Year",
     type = "o", xlim = c(1990, 2021), ylim = c(min(maxtemp_1990), max(maxtemp_1990)),
     main = "SES vs Damped Holt Forecast of Melbourne Max Annual Temperature")

# Add SES fitted and forecasted values (blue)
lines(fitted(fit_ses), col = "blue", type = "o")
lines(fit_ses$mean, col = "blue", type = "o")

# Add Holt fitted and forecasted values (red)
lines(fitted(fit_holt), col = "red", type = "o")
lines(fit_holt$mean, col = "red", type = "o")

# Add legend at top-left
legend("topleft",
       legend = c("Observed Data",
                  "SES (Fitted + Forecast)",
                  "Damped Holt (Fitted + Forecast)"),
       col = c("black", "blue", "red"),
       lty = 1, pch = 1, bty = "n")


# Compare AICc and BIC of SES and Holt models

# Refit SES using ets() to get AICc and BIC
fit_ses2 = ets(maxtemp_1990, model = "ANN")

# Refit Damped Holt using ets() (AAN = Additive trend, No seasonality)
fit_holt2 = ets(maxtemp_1990, model = "AAN", damped = TRUE)

# Create a small comparison table
AIC_BIC_Compare = data.frame(
  Model = c("SES (ANN)", "Damped Holt (AAN)"),
  AIC = c(fit_ses2$aic, fit_holt2$aic),
  AICc = c(fit_ses2$aicc, fit_holt2$aicc),
  BIC = c(fit_ses2$bic, fit_holt2$bic)
)
